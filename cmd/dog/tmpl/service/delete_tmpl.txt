	{{.PackageName}}Model := model.New{{.ModelName}}Model()

	query := "id IN ?"

	if err != nil {
		return fmt.Errorf("model count {{.PackageName}} err: %w", err)
	}

	if int(total) != len(in.IDS) {

	}

	}
